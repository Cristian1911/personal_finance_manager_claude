Replace the entire contents of `webapp/src/lib/supabase/admin.ts` with the following:

```typescript
import { createClient as createSupabaseClient } from "@supabase/supabase-js";
import type { Database } from "@/types/database";

export function createAdminClient() {
  const serviceRoleKey = process.env.SUPABASE_SECRET_KEY;

  if (!serviceRoleKey) {
    throw new Error(
      "Admin client unavailable: SUPABASE_SECRET_KEY not set"
    );
  }

  return createSupabaseClient<Database>(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    serviceRoleKey,
    {
      auth: {
        persistSession: false,
        autoRefreshToken: false,
      },
    }
  );
}
```

The return type is now inferred as `SupabaseClient<Database>` (never null) — no explicit return type annotation needed. TypeScript will infer the non-null type automatically because the only code path that doesn't throw returns the client.
