    **1. Create `webapp/src/lib/health-score.ts`**:

    This file computes the composite 0-100 score from 4 health meter dimensions (per D-09: equal-weight average, 25% each).

    ```typescript

    import type { MeterType } from "@/lib/health-levels";

    /**
     * Convert a raw meter value to a 0-100 score where 100 = best.
     * Each meter type has different scales and directions.
     */
    export function meterToScore(type: MeterType, value: number): number { ... }

    /**
     * Compute composite health score (0-100) from 4 meter dimensions.
     * Equal weight: 25% each. Returns 0-100 rounded to nearest integer.
     * Meters without data (hasData=false) are excluded from average.
     */
    export function computeCompositeScore(meters: HealthMeter[]): number { ... }

    /**
     * Get the Spanish label for a composite score range.
     * Per UI-SPEC: 0-25 Critico, 26-50 Atento, 51-75 Bien, 76-90 Solido, 91-100 Excelente
     */
    export function getScoreLabel(score: number): string { ... }

    /**
     * Get the CSS color variable for a score value.
     */
    export function getScoreColor(score: number): string { ... }
    ```

    **meterToScore implementation details:**
    - `gasto`: Lower is better. Score = `Math.max(0, Math.min(100, 100 - value))`. At 0% expense ratio = 100 score, at 100% = 0 score.
    - `deuda`: Lower is better. Score = `Math.max(0, Math.min(100, 100 - (value / 60) * 100))`. At 0% DTI = 100, at 60% DTI = 0. 60% is the meaningful max (well beyond critico at 43%).
    - `ahorro`: Higher is better. Score = `Math.max(0, Math.min(100, (value / 40) * 100))`. At 0% savings = 0, at 40%+ = 100.
    - `colchon`: Higher is better. Score = `Math.max(0, Math.min(100, (value / 15) * 100))`. At 0 months = 0, at 15+ months = 100.

    **computeCompositeScore:**
    - Filter meters where `hasData === true`
    - If no meters have data, return 0
    - Compute average of `meterToScore(m.type, m.value)` for each valid meter
    - Round to nearest integer

    **getScoreLabel:**
    - 0-25: "Critico"
    - 26-50: "Atento"
    - 51-75: "Bien"
    - 76-90: "Solido"
    - 91-100: "Excelente"

    **getScoreColor:**
    - 0-25: "var(--z-debt)"
    - 26-50: "var(--z-expense)"
    - 51-75: "var(--z-income)"
    - 76-100: "var(--z-excellent)"

    ```tsx
    export function HealthScoreSection({ data }: HealthScoreSectionProps) {
      const [expandedMeter, setExpandedMeter] = useState<MeterType | null>(null);
      const compositeScore = computeCompositeScore(data.meters);
      const scoreLabel = getScoreLabel(compositeScore);

      const selectedMeter = expandedMeter
        ? data.meters.find((m) => m.type === expandedMeter) ?? null
        : null;

      return (
        <div className="rounded-xl bg-z-surface-2 py-6 px-4 space-y-4">
          {/* Speedometer gauge */}
          <SpeedometerGauge score={compositeScore} label={scoreLabel} />

          {/* Individual meter rows */}
          <div className="space-y-1">
            {data.meters.map((meter) => (
              <MeterRow
                key={meter.type}
                meter={meter}
                onClick={() => setExpandedMeter(meter.type)}
              />
            ))}
          </div>

          {/* Summary roast */}
          <div
            className="rounded-lg px-3 py-2.5 text-xs"
            style={{ backgroundColor: `color-mix(in srgb, ${getLevelColor(worstMeter.level)} 15%, transparent)` }}
          >
            <span style={{ color: summaryColor }} className="font-semibold">
              {data.summaryRoast}
            </span>
          </div>

          {/* Expanded detail sheet */}
          {selectedMeter && (
            <HealthMeterExpanded
              meter={selectedMeter}
              income={data.monthlyIncome}
              currency={data.currency}
              open={expandedMeter !== null}
              onOpenChange={(open) => { if (!open) setExpandedMeter(null); }}
            />
          )}
        </div>
      );
    }
    ```
